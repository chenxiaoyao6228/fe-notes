## 前言

实现一个红绿灯，支持以下功能：

- 按照红灯->绿灯->黄灯的方式循环打印, 打印时间间隔分别为 3s->2s->1s

## 实现: 借助 async-await

```js
f function renderLight(color) {
        const lights = document.querySelectorAll(".light");
        lights.forEach(
          (light) => (light.style.backgroundColor = "transparent")
        );
        document.querySelector(`.${color}`).style.backgroundColor = color;
      }

      function sleep(ms) {
        return new Promise((resolve) => setTimeout(resolve, ms));
      }

      async function renderLightWidthDelay(color, duration) {
        renderLight(color);
        await sleep(duration);
      }

      const lightRunner = (() => {
        const tasks = [];
        let curIndex = 0;

        const add = (task) => {
          tasks.push(task);
        };

        const run = async () => {
          while (tasks.length) {
            const task = tasks[curIndex];
            await task();
            if (curIndex >= tasks.length - 1) {
              curIndex = 0;
            } else {
              curIndex++;
            }
          }
        };

        return {
          add,
          run,
        };
      })();

      lightRunner.add(() => renderLightWidthDelay("red", 1000));
      lightRunner.add(() => renderLightWidthDelay("green", 1000));
      lightRunner.add(() => renderLightWidthDelay("yellow", 1000));
      lightRunner.run();
```
